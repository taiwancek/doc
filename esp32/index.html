叫裝置切換開關 ¶

//topic: {uuid}
//msg:
{
    "cmd":"switch",
    "data": {
        "pin":"a",
        "value":0 //0:close[low], 1:open[high]
    }
}

//topic(裝置回覆): {uuid}-response
//msg:
{
    "cmd":"switch",
    "data": {
        "pin":"a",
        "value":0 //0:close[low], 1:open[high]
    }
    "state": 200 //200:成功,400:失敗
}
